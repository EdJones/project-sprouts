def create_outer_task
  t = task pkg_name
  # This helps executable rake tasks decide if they
  # want to do something special for library tasks.
  t.sprout_entity = self
  class << t
    def sprout_library?
      !sprout_entity.nil? &&
        sprout_entity.is_a?(Sprout::Library)
    end
  end
  t
end
